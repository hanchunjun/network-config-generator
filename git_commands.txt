# Git 上传命令集

# 1. 初始化Git仓库
git init

# 2. 添加所有文件
git add .

# 3. 首次提交
git commit -m "Initial commit: 网络设备配置脚本生成工具 V1.0.0 测试版"

# 4. 关联远程仓库（请替换为你的GitHub仓库地址）
git remote add origin https://github.com/your-username/network-config-generator.git

# 5. 推送到main分支
git push -u origin main
